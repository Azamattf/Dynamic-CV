% Style definitions for CV sections

% Section header style
\newcommand{\sectionheader}[1]{%
  \vspace*{0.5cm}%
  \begin{tikzpicture}
    \node[
      fill=\primarycolor,
      text=white,
      font=\Large\bfseries\sffamily,
      inner sep=0.3cm,
      minimum width=\columnwidth,
      anchor=west
    ] {#1};
  \end{tikzpicture}%
  \vspace*{0.3cm}%
}

% Overflow section header style
\newcommand{\overflowsectionheader}[1]{%
  \vspace*{0.5cm}%
  \begin{tikzpicture}
    \node[
      fill=\primarycolor!80,
      text=white,
      font=\Large\bfseries\sffamily,
      inner sep=0.3cm,
      minimum width=\columnwidth,
      anchor=west
    ] {#1 \overflowsuffix};
  \end{tikzpicture}%
  \vspace*{0.3cm}%
}

% Personal info item style
\newcommand{\infoitem}[2]{%
  \begin{tikzpicture}
    \node[anchor=west, text=\primarycolor, minimum width=1.5em] (icon) {#1};
    \node[anchor=west, text width=\columnwidth-2em, align=left, right=0.2em of icon] {#2};
  \end{tikzpicture}\\[0.2cm]%
}

% Skill bar style
\newcommand{\skillbar}[2]{%
  \begin{tikzpicture}
    \node[anchor=west, text width=0.4\columnwidth, align=left] (skill) {#1};
    \begin{scope}[xshift=0.4\columnwidth]
      \draw[fill=\primarycolor!20, rounded corners=1pt] (0,0) rectangle (0.55\columnwidth,0.3cm);
      \draw[fill=\primarycolor, rounded corners=1pt] (0,0) rectangle (#2*0.55\columnwidth,0.3cm);
    \end{scope}
  \end{tikzpicture}\\[0.2cm]%
}

% Timeline entry style for work experience & education
\newcommand{\timelineentry}[5]{%
  % #1: Start date
  % #2: End date
  % #3: Role/Title
  % #4: Company/Institution
  % #5: Description
  \begin{minipage}{\columnwidth}
    \textbf{\large #3}\\
    \textcolor{\secondarycolor}{\textbf{#4}}\\
    \textit{#1 -- #2}\\[0.1cm]
    #5\\[0.3cm]
  \end{minipage}%
}

% Award style
\newcommand{\awardentry}[3]{%
  % #1: Year
  % #2: Award name
  % #3: Awarded by
  \begin{minipage}{\columnwidth}
    \textbf{#2} (\textit{#1})\\
    \textcolor{\tertiarycolor}{#3}\\[0.3cm]
  \end{minipage}%
}

% Publication style
\newcommand{\publicationentry}[4]{%
  % #1: Year
  % #2: Title
  % #3: Journal/Conference
  % #4: Authors
  \begin{minipage}{\columnwidth}
    \textbf{#2} (\textit{#1})\\
    \textit{#3}\\
    #4\\[0.3cm]
  \end{minipage}%
}

% Project style
\newcommand{\projectentry}[3]{%
  % #1: Year
  % #2: Project name
  % #3: Description
  \begin{minipage}{\columnwidth}
    \textbf{#2} (\textit{#1})\\
    #3\\[0.3cm]
  \end{minipage}%
}

% Summary style
\newcommand{\summaryentry}[1]{%
  \begin{minipage}{\columnwidth}
    \textit{#1}\\[0.2cm]
  \end{minipage}%
}
